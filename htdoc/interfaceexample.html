A Package with Interface and Implementation Parts

The following two packages provide the same support as is provided by the single export package. 

%% File perms.ast

Package perms

%% This example package can be imported.  It is 
%% contains only the interface information.

=======================================================================
			interface
=======================================================================

Expect
  perms		: [`a] -> [[`a]]

		%" perms(x) is a list of all of the permutations
		%" of list x.  For example, perms("abc") =
		%" ["abc", "acb", "bac", "bca", "cab", "cba"].
%Expect

%Package



%% File perms.asi

Package permsImplementation

%% This package cannot be imported directly.  You import perms.ast
%% instead.  This package just provides the implementation.

=======================================================================
			implements "perms"
=======================================================================

%% We need the listpat package.  This is imported in the
%% implementation part because it is only needed for the
%% implementation.

Import "pattern/listpat".

%% Here is the implementation of perms.

Define perms by first
  case perms []   = [[]]
  case perms (?z) = [: a :: p | Match ?x ++ [?a] ++ ?y = z.
                                Let p = each(perms(x++y)). :]
%Define

%Package
